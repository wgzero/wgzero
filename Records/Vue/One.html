# III.Vue使用WebSocket

# 1.vue create demo

# 2.在views下创建Ws.vue文件

<template>
  <div>
      <h1>websocket</h1>
      <div class="ws">
          <div>編號: {{ list.userid }}</div>
          <div>隊伍: {{ list.team }}</div>
          <div>名字: {{ list.userName }}</div>
      </div>
  </div>
</template>

<script>
export default {
  props: {},
  data() {
    return {
        websock: '',
        list: ''
    };
  },
  computed: {},
  mounted() {
    this.initWebSocket(); //页面渲染的时候，对ws进行初始化
  },
  methods: {
    initWebSocket() {
        // demo
      this.websock = new WebSocket("ws://localhost:8686"); //这个连接ws://固定，后面的根据自己的IP和端口进行改变，我设置监听的就是8686
      // 1.建立通信，获取后端返回数据
      this.websock.onmessage = this.websocketonmessage;
      // 2.通信失败提示
      this.websock.onerror = this.websocketonerror;
      // 3.数据信息传递，向后端传递数据
      this.websock.onopen = this.websocketonopen;
      // 4.通信关闭
      this.websock.onclose = this.websocketclose;
      console.log('ws', this.websock)
    },
    websocketonopen() {
      // 连接建立之后执行send方法发送数据，这个和自己的后端沟通好需要传什么数据，我的是要进行token验证
      let token = sessionStorage.getItem("token");
      let data = {
        type: "REGISTER",
        data: { token: token }
      };
      // this.websock.send(JSON.stringify(data));
      this.websock.send("connection");
    },
    websocketonerror() {
      //连接错误
      console.log("WebSocket连接失败");
    },
    websocketonmessage(e) {
      // 数据接收
      console.log('e', e.data)
      let res = JSON.parse(e.data)
      this.list = res.data
      console.log('list', this.list) 
    //   console.log(JSON.parse(e.data));
    },
    websocketclose(e) {
      // 关闭连接
      console.log("已关闭连接", e);
    }
  },
  destroyed() {
    this.websock.close(); // 页面销毁后断开websocket连接
  }
};
</script>
