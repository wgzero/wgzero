# 2.防抖(debounce)和节流(throttle)

<div>
    <button id="btn">点击</button>
</div>
<script>
    // 防抖：多次点击只保留最后一次点击有效
    // 函数,时间
    function debounce(fn ,delay){
        let timer = null;
        return function(){
            // 清除timer
            clearTimeout(timer)
            timer = setTimeout(function(){
                // debugger
                fn.apply(this)
            }, delay)
        }
    }

    document.getElementById('btn').onclick = debounce(function(){
        console.log('点击事件被触发了...:' + Date.now())
    }, 1000)
</script>

// 节流：一段时间内只执行一次
function throttle(fn, delay){
    var startTime = 0;
    return function(){
        var endTime = Date.now();
        if(endTime - startTime > delay){
            fn.call(this);
            startTime = endTime;
        }
    }
}
document.onscroll = throttle(function(){
    console.log('被触发了...:' + Date.now());
}, 1000)

// 数组去重
// 1.Array.from(new Set())
let arr = [1,1,2,2,3,4,5,5,6,6,7,7,8,9,1,2,3,4,5];
let a1 = Array.from(new Set(arr));
console.log('a1', a1);

// 2.indexOf处理
function sRes(num){
    let arrx = []
    for(let i=0; i < num.length ;i++){
        if(arrx.indexOf(num[i]) == -1){
            arrx.push(num[i])
        }
    }
    return arrx;
}
let a2 = sRes(arr);
console.log('a2', a2);

// 3.splice实现
let resArr = function(num){
    for(let i=0; i < num.length;i++){
        for(let j = i+1;j < num.length;j++){
            if(num[i] == num[j]){
                num.splice(j, 1)
            }
        }
    }
    return num;
}
let a3 = resArr(arr);
console.log('a3', a3);

// 4.sort排序比较相邻的两个数去重
let resArr1 = function(num){
    let z = [];
    num = num.sort();
    for(let i=0; i < num.length ;i++){
        // if(num[i] == num[i+1]){
        //     num.splice(i, 1)
        // }
        if(num[i] !== num[i+1]){
            z.push(num[i])
        }
    }
    return z;
}

let a4 = resArr1(arr);
console.log('a4', a4);

// 5.利用对象的属性不能相同的特点进行去重
let resArr2 = function(num){
    let obj = {};
    let arrS = [];
    for(let i=0; i < num.length;i++){
        if(!obj[num[i]]){
            arrS.push(num[i])
            obj[num[i]] = 1
        } else {
            obj[num[i]]++
        }
    }

    return arrS;
}

let a5 = resArr2(arr);
// console.log('a5', a5)

let resArr3 = function(num){
    let arr = []
    for(let i=0;i<num.length;i++){
        if(!arr.includes(num[i])){
            arr.push(num[i])

    return arr;

let a6 = resArr3(arr);

// 7.利用hasOwnProperty
let resArr4 = function(num){
    let obj = {};
    return num.filter(function(item, index){
        return obj.hasOwnProperty(typeof item + item) ? false : (obj[typeof item +item]) = true;

# 5.时间格式化

// 时间格式化
// 1.获取到当前时间戳
let d = Date.now();
console.log('d', d); // 1592365019217
// 处理时间格式：年月日+年月日时分秒
/*
A.传入时间戳
B.传入年月日的格式：
C.传入时分秒的格式：
D.判断是否显示年月日还是年月日时分秒，默认是年月日时分秒
*/ 
function handleFormatTimes(num, p1, p2, p3 = true){
    let res = new Date();
    // y m d h m s 2020-06-01 06:04:04
    let y = res.getFullYear(num)
    let mm = res.getMonth() + 1
    let d = res.getDate()
    let h = res.getHours()
    let m = res.getMinutes()
    let s = res.getSeconds()

    // 前缀要不要小于10前面加0
    mm = mm > 10 ? mm : '0'+ mm;
    d = d > 10 ? d : '0'+ d;
    h = h > 10 ? h : '0'+ h;
    m = m > 10 ? m : '0'+ m;
    s = s > 10 ? s : '0'+ s;

    let time = p3 ? y + p1 + mm + p1 + d + ' ' + h + p2 + m + p2 + s :  y + p1 + mm + p1 + d + ' '
    return time;
}

let aa = handleFormatTimes(d, '-', ':')
console.log('aa', aa); // 2020-06-17 11:36:59

# 7.函数柯里化

// 函数颗粒化：就是接收多个参数的函数变换成一个单一参数的函数
// add(1)(2)(3) = 6
// add(1)(2)(3)(4) = 10
// add(1)(2)(3)(4)(5) = 15

function add(){
    // A.第一次执行时，定义一个数组专门用来存储所有的参数
    let _args = Array.prototype.slice.call(arguments) 
    // B.在内部声明一个函数，利用闭包的特性保存_args并收集所有的参数值
    let _adder = function(){
        _args.push(...arguments)
        return _adder
    }
    // 利用toString隐式转换的特性，当最后执行时隐式转换，并计算最终的值返回
    _adder.toString = function(){
        return _args.reduce(function(a, b){
            return a + b
        })
    }
    return _adder;
}

let a1 = add(1)(2)(3)
console.log('a1', a1) // 6
let a2 = add(1)(2)(3)(4)
console.log('a2', a2) // 10
let a3 = add(1)(2)(3)(4)(5)
console.log('a3', a3) // 15

# 8.多维数组转成一维数组

let arr = [1,[2,[3,[4,[5,6,7]]]]];
// 法一：多维转一维
console.log(arr.flat(Infinity));

// 法二： 二维转一维
console.log([].concat.apply([],arr));

//法三：实现一个扁平化函数  递归： 将二维转一维递归
function myFlat(arr){
    let arrs = [];
    arr.forEach((v) => {
        if(Array.isArray(v)){
            arrs = arrs.concat(myFlat(v));
        }else{
            arrs.push(v);
        }
    })
    return arrs;
}
console.log(myFlat(arr));
